ring <- read.table("ring.txt")
mu1 <- c(1, 1)
mu2 <- c(-5, -5)
sigma <- diag(2)


x <- seq(min(ring[, 1]), max(ring[, 1]), by = .1)
y <- seq(min(ring[, 2]), max(ring[, 2]), by = .1)
net1 <- outer(x, y, function(x, y) dmvnorm(cbind(x, y), mu1, sigma))
net2 <- outer(x, y, function(x, y) dmvnorm(cbind(x, y), mu2, sigma))
plot(ring)
# plot()
contour(x, y, net1, add = T)
contour(x, y, net2, add = T)
